from flask import Flask, render_template, request, redirect, url_for app = Flask(__name__) @app.route('/') def index(): return render_template('login.html') @app.route('/login', methods=['POST']) def login(): username = request.form['username'] password = request.form['password'] # Here, you'd typically check credentials against a database. # For this example, let's assume valid login. if username == 'username' and password == 'password': return redirect(url_for('success')) else: return redirect(url_for('index')) @app.route('/success') def success(): return 'Login successful!' if __name__ == '__main__': app.run(debug=True)
